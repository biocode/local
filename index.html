Simple Ajax to services and functions
Promises-based, with streaming, piping, and automatic body-parsing.

// Standard Ajax
web.get('https://myhost.com', { query: 'params' })
  .accept('json')
  .then(function(res) {
    res.status; // => 200
    res.reason; // => 'OK'
    res.ContentType; // => 'application/json'
    res.body; // => { hello: 'world' }
  });

// Emulated HTTP for local requests
web.at('#hi', function(req, res) {
  res.status(200, 'OK');
  res.contentType('html');
  res.link({ href: '#hi', rel: 'self service foo.com/hello', title: 'Hello Service' });
  res.end('<h1>Hello, '+req.params.who+'</h1>');
});
web.get('#hi', { who: 'World' }).then(...);

Communicate across postMessage
Iframes, workers, and other postMessage() channels carry client-side requests.

// Workers
var myWorker = new Worker('/my-worker.js');
web.at('#my-worker.js', myWorker);
web.get('#my-worker.js', { who: 'Worker' }).then(...);
	
// Frames
web.at('#parent', window.parent);
web.at('#my-iframe', document.querySelector('iframe#my-iframe'));

Standardize and Simplify with Search
Link headers and document <link> elements construct URLs for you.

// Build a set of links to search
web.addLinks('#parent');              // add links from parent frame
web.addLinks('#my-worker.js');        // add links from worker
web.addLinks(window.location.origin); // add links from host
web.addLinks(document);               // add document links

// Search all links for a foo.com/hello service
web.get({ rel: 'foo.com/hello', who: 'Consumer' }).then(function(res) {
    res.body; // => <h1>Hello, Consumer</ht>
});
	
// Chain searches across navigations
web.head({ rel: 'foo.com/crud', id: 'users' }).get({ rel: 'item', id: 'bob' })
